    'use strict';

const Duration = require('./Duration');

Object.defineProperty(Number.prototype, "seconds", {
  get: function() {
    return new Duration(this.valueOf());
  }
});

Object.defineProperty(Number.prototype, "second", {
  get: function() {
    return this.seconds;
  }
});
